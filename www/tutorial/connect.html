1. Connect over wss://

Point any SpacetimeDB-compatible client at a public relay frontend. The host is the TLS endpoint; the database name is the mirror module, not the upstream BitCraft database.

  1. Choose region and mirror name From /health (or the home page table), take port and database. Example for region 14: host wss://relay.bitcraftsync.app:3014, database relay-mirror-bc14.
  2. Offer a subprotocol Prefer v2.bsatn.spacetimedb for current SDKs. Offer v1.bsatn.spacetimedb when you need full v1 TransactionUpdates (the frontend synthesises those from local TULs). The listener echoes the chosen protocol.
  3. Ask for uncompressed frames Use ?compression=None on the subscribe URL. The relay does not decompress inbound Brotli/Gzip payloads for you.
  4. Wait for InitialConnection After the WebSocket opens, the first binary server message is InitialConnection (identity + token). Persist the token if your SDK supports reconnect-as-same-identity.

Addresses

UseURL

Handshake header Sec-WebSocket-Protocol: v2.bsatn.spacetimedb
No auth token required for public read subscriptions on these mirrors. Do not send your BitCraft upstream bearer token to the relay — that token is only for the relay process itself talking to BitCraft.

Python

Two workable paths: the official SpacetimeDB Python SDK (codegen bindings + async client), or a thin websockets client if you already speak BSATN. Most tools should use the SDK after generating bindings from the schema endpoint. 

# pip install spacetimedb-sdk
# Generate bindings from the relay schema first (see "Discover schema"):
#   spacetime generate --lang python \
#     --out-dir module_bindings \
#     --module-name relay_mirror_bc14 \
#     # or feed the downloaded schema JSON to your codegen pipeline

import asyncio
from spacetimedb_sdk.spacetimedb_async_client import SpacetimeDBAsyncClient
import module_bindings

HOST = "relay.bitcraftsync.app:3014"   # host:port only
DATABASE = "relay-mirror-bc14"
SSL = True

client = SpacetimeDBAsyncClient(module_bindings)

def on_connect(_client, identity, token):
    print(f"connected identity={identity} token_len={len(token or '')}")

async def main():
    # auth_token=None → anonymous identity issued by local stdb
    await client.run(
        None,
        f"https://{HOST}",
        DATABASE,
        on_connect,
        [],  # subscribe in a later step — or pass initial SQL here
        ssl_enabled=SSL,
    )

asyncio.run(main())

# Prefer the SDK unless you already have a BSATN codec.

import asyncio
import websockets

SUBPROTOCOL = "v2.bsatn.spacetimedb"
URL = (
    "wss://relay.bitcraftsync.app:3014"
    "/v1/database/relay-mirror-bc14/subscribe"
    "?compression=None"
)

async def main():
    async with websockets.connect(
        URL,
        subprotocols=[SUBPROTOCOL],
        max_size=None,  # BitCraft tables can be huge
    ) as ws:
        # First binary frame: compression_tag (0) + BSATN InitialConnection
        frame = await ws.recv()
        assert isinstance(frame, (bytes, bytearray))
        assert frame[0] == 0, "expected compression=None"
        print(f"got {len(frame)} byte handshake frame")
        # Decode BSATN ServerMessage::InitialConnection with your codec…
        await asyncio.Future()  # keep alive

asyncio.run(main())

Rust

Official spacetimedb-sdk after spacetime generate is the usual path. Below: SDK builder, then the same wire shape relay-cache uses internally (useful when you subscribe with raw SQL strings and no generated module). 

// After: spacetime generate --lang rust -o module_bindings …
use module_bindings::DbConnection;
use spacetimedb_sdk::DbContext;

fn main() {
    let conn = DbConnection::builder()
        .with_uri("wss://relay.bitcraftsync.app:3014")
        .with_database_name("relay-mirror-bc14")
        .on_connect(|_ctx, identity, token| {
            println!("connected {identity} token_len={}", token.len());
        })
        .on_connect_error(|_ctx, err| eprintln!("connect error: {err}"))
        .on_disconnect(|_ctx, err| {
            if let Some(e) = err {
                eprintln!("disconnected: {e}");
            }
        })
        .build()
        .expect("connect");

    // Drive the SDK event loop however your app does (tick / block_on).
    let _ = conn;
}
use anyhow::{anyhow, Result};
use futures_util::StreamExt;
use http::header::SEC_WEBSOCKET_PROTOCOL;
use tokio_tungstenite::tungstenite::client::IntoClientRequest;
use tokio_tungstenite::tungstenite::Message;

const SUBPROTOCOL: &str = "v2.bsatn.spacetimedb";

#[tokio::main]
async fn main() -> Result<()> {
    let url = concat!(
        "wss://relay.bitcraftsync.app:3014",
        "/v1/database/relay-mirror-bc14/subscribe",
        "?compression=None",
    );
    let mut req = url.into_client_request()?;
    req.headers_mut()
        .insert(SEC_WEBSOCKET_PROTOCOL, SUBPROTOCOL.parse()?);

    let (mut ws, _) = tokio_tungstenite::connect_async(req).await?;
    let Some(Ok(Message::Binary(data))) = ws.next().await else {
        return Err(anyhow!("expected binary InitialConnection"));
    };
    assert_eq!(data[0], 0, "compression tag");
    // bsatn::from_slice::<ServerMessage>(&data[1..]) → InitialConnection
    println!("handshake {} bytes", data.len());
    Ok(())
}
